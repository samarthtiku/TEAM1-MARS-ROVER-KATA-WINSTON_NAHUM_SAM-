12       @Override
13       public void processCommands(String commands) {
14           for (char command : commands.toUpperCase().replaceAll("\\s", "").toCharArray()) {
15               switch (command) {
16                   case 'L':
17                       turnLeft();
18                       break;
19                   case 'R':
20                       turnRight();
21                       break;
22                   case 'M':
23                       move();
24                       break;
25                   default:
26                       throw new IllegalArgumentException("Invalid command: " + command);
27               }
28           }
29       }
30   
31       @Override
32       protected void move() {
33           int newX = x + (orientation == Orientation.E ? 1 : (orientation == Orientation.W ? -1 : 0));
34           int newY = y + (orientation == Orientation.N ? 1 : (orientation == Orientation.S ? -1 : 0));
35   
36           if (plateau.isValidPosition(newX, newY) && !plateau.isOccupied(newX, newY)) {
37               plateau.removeRover(x, y);
38               x = newX;
39               y = newY;
40               plateau.placeRover(x, y, id);
41               distanceTraveled++;
42           } else {
43               System.out.println("Warning: Rover " + id + " cannot move. Position blocked or out of bounds.");
44           }
45       }
